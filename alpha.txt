-- create a new table called "customers"
CREATE TABLE customers (
  id INT PRIMARY KEY,
  name VARCHAR(50),
  email VARCHAR(50),
  country VARCHAR(50)
);

-- insert some sample data into the "customers" table
INSERT INTO customers (id, name, email, country)

  (2, 'Jane Smith', 'jane.smith@example.com', 'Canada'),
  (3, 'Bob Johnson', 'bob.johnson@example.com', 'USA'),
  (4, 'Alice Lee', 'alice.lee@example.com', 'Australia');

-- retrieve all customers from the USA
SELECT *
FROM customers
WHERE country = 'USA';
